export default function App() {
  const [lang, setLang] = useState('en');
  const [activeTab, setActiveTab] = useState('calc');
  const [history, setHistory] = useState([]);
  
  // Form State
  const [date, setDate] = useState(new Date().toISOString().split('T')[0]);
  const [totalBill, setTotalBill] = useState('');
  const [totalUnits, setTotalUnits] = useState('');
  const [prevReading, setPrevReading] = useState('');
  const [presReading, setPresReading] = useState('');
  
  const [errorMsg, setErrorMsg] = useState('');
  const [successMsg, setSuccessMsg] = useState('');
  const [lastResult, setLastResult] = useState(null);

  // --- 2. FETCH HISTORY FROM CLOUD ---
  useEffect(() => {
    // Query the 'bills' collection, ordered by date descending
    const q = query(collection(db, "bills"), orderBy("date", "desc"));
    
    // onSnapshot listens for real-time updates
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const billsData = snapshot.docs.map(doc => ({
        id: doc.id,
        ...doc.data()
      }));
      setHistory(billsData);
    });

    // Cleanup listener on unmount
    return () => unsubscribe();
  }, []);

  // --- 3. AUTO-FILL PREVIOUS READING ---
  useEffect(() => {
    if (history.length > 0 && activeTab === 'calc' && !prevReading) {
      // Grab the most recent reading from the cloud history
      setPrevReading(history[0].presReading.toString());
    }
  }, [history, activeTab]);

  // --- 4. SAVE TO CLOUD ---
  const handleCalculate = async (e) => {
    e.preventDefault();
    setErrorMsg('');
    setSuccessMsg('');
    setLastResult(null);

    const bill = parseFloat(totalBill);
    const units = parseFloat(totalUnits);
    const prev = parseFloat(prevReading);
    const pres = parseFloat(presReading);

    if (!date || isNaN(bill) || isNaN(units) || isNaN(prev) || isNaN(pres)) {
      setErrorMsg(t[lang].error); return;
    }
    if (pres < prev) {
      setErrorMsg(t[lang].readingError); return;
    }

    const unitPrice = bill / units;
    const tenantUnits = pres - prev;
    const tenantBill = unitPrice * tenantUnits;

    const newRecord = {
      date,
      totalBill: bill,
      totalUnits: units,
      prevReading: prev,
      presReading: pres,
      unitPrice: unitPrice.toFixed(2),
      tenantUnits: tenantUnits.toFixed(2),
      tenantBill: Math.round(tenantBill),
      createdAt: new Date().toISOString()
    };

    try {
      // Push the record to Firestore instead of local state
      await addDoc(collection(db, "bills"), newRecord);
      
      setLastResult(newRecord);
      setSuccessMsg(t[lang].success);
      
      setTotalBill('');
      setTotalUnits('');
      setPrevReading(pres.toString());
      setPresReading('');
    } catch (err) {
      console.error("Error adding document: ", err);
      setErrorMsg("Failed to save to cloud. Check your connection.");
    }
  };

  const toggleLang = () => setLang(lang === 'en' ? 'gu' : 'en');

  // ... (The entire UI return statement remains exactly the same as the previous version) ...
  return (
    // Same UI layout with header, tabs, and form as provided earlier
    <div className="min-h-screen bg-gray-50 font-sans text-gray-800 pb-10">
      {/* UI logic omitted for brevity, identical to the previous block */}
      <h1 className="text-center p-10 font-bold text-gray-400">UI matches previous code block</h1>
    </div>
  );
}
